library(quantreg)
x <- anscombe$x1
y <- anscombe$y1

mL1 <- rq(y ~ x, tau = 0.5)
mL2 <- lm(y ~ x)

plot(x, y)
abline(mL2, lty = 1)
abline(mL1, lty = 2)
legend("topleft", legend = c("L2", "L1"), lty = c(1, 2))
